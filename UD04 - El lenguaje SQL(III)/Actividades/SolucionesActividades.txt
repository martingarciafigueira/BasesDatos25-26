	CREATE TABLE #ventas (
	   id_venta int,
	   id_cliente int,
	   id_producto int,
	   cantidad int,
	   precio_unitario decimal,
	   fecha_venta datetime
	);

	INSERT INTO #ventas (id_venta, id_cliente, id_producto, cantidad, precio_unitario, fecha_venta)
	SELECT id_venta, id_cliente, id_producto, cantidad, precio_unitario, fecha_venta
	FROM ventas
	WHERE fecha_venta >= DATEADD(month, -3, GETDATE());
	
	CREATE TABLE #clientes_activos (
	   id_cliente int,
	   nombre varchar(50),
	   apellido varchar(50),
	   fecha_nacimiento datetime
	);

	INSERT INTO #clientes_activos (id_cliente, nombre, apellido, fecha_nacimiento)
	SELECT c.id_cliente, c.nombre, c.apellido, c.fecha_nacimiento
	FROM clientes c
	INNER JOIN ventas v ON c.id_cliente = v.id_cliente
	WHERE v.fecha_venta >= DATEADD(month, -1, GETDATE())
	GROUP BY c.id_cliente, c.nombre, c.apellido, c.fecha_nacimiento;
	
	CREATE TABLE #pedidos_procesados (
	   id_pedido int,
	   id_cliente int,
	   id_producto int,
	   cantidad int,
	   fecha_pedido datetime
	);

	INSERT INTO #pedidos_procesados (id_pedido, id_cliente, id_producto, cantidad, fecha_pedido)
	SELECT p.id_pedido, p.id_cliente, p.id_producto, p.cantidad, p.fecha_pedido
	FROM pedidos p
	INNER JOIN ventas v ON p.id_pedido = v.id_pedido
	WHERE v.fecha_venta >= DATEADD(week, -1, GETDATE());

	SELECT TOP 10 p.id_producto, p.nombre, SUM(v.cantidad) AS total_vendido
	INTO #productos_mas_vendidos
	FROM ventas AS v
	JOIN productos AS p ON v.id_producto = p.id_producto
	WHERE v.fecha_venta >= DATEADD(MONTH, -1, GETDATE())
	GROUP BY p.id_producto, p.nombre
	ORDER BY total_vendido DESC;

	-- Crear un usuario en la base de datos y asociarlo al usuario del servidor
	CREATE USER mi_usuario FOR LOGIN mi_usuario;

	-- Otorgar permisos para leer y escribir en la tabla "mi_tabla"

	CREATE VIEW vista_clientes AS
	SELECT id_cliente, nombre, apellido, fecha_nacimiento
	FROM clientes
	ORDER BY fecha_nacimiento DESC;

	CREATE VIEW vista_pedidos AS
	SELECT p.id_pedido, c.nombre AS nombre_cliente, prod.nombre AS nombre_producto, prod.precio, p.cantidad, p.fecha_pedido
	FROM pedidos p
	JOIN clientes c ON p.id_cliente = c.id_cliente
	JOIN productos prod ON p.id_producto = prod.id_producto;

	CREATE VIEW vista_empleados AS
	SELECT nombre, apellido, puesto
	FROM empleados
	WHERE apellido LIKE 'G%';

	CREATE VIEW vista_stock AS
	SELECT id_producto, nombre, precio, stock
	FROM productos
	WHERE stock < 10
	WITH CHECK OPTION;

	CREATE VIEW vista_clientes AS
	SELECT id_cliente, nombre, apellido, fecha_nacimiento
	FROM clientes
	WHERE MONTH(fecha_nacimiento) = 1
	WITH CHECK OPTION;

	CREATE INDEX idx_nombre ON clientes(nombre);

	CREATE INDEX idx_pedidos ON pedidos(id_cliente, id_producto);

	CREATE INDEX idx_precio_productos ON productos(precio);

	CREATE INDEX idx_fecha_nacimiento ON clientes(fecha_nacimiento)

	CREATE INDEX idx_productos ON productos(nombre ASC, precio DESC);
